#+BEGIN_SRC sh
doveconf -n | head -n1
#+END_SRC

2. copy files from location to /etc/dovecot/
#+BEGIN_SRC sh
cp -r /usr/share/doc/dovecot/example-config/* /etc/dovecot/
#+END_SRC

3. Authentication
#+BEGIN_SRC sh
echo "$USER:{PLAIN}password:$UID:$GID::$HOME" > users
sudo mv users /etc/dovecot/
